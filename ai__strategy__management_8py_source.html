582  def __getMovementArrayFromBroadcast(self, broadCastIndex: int):
583  """This function is used by AI to know which movements to do to reach its teammate
584  return : movement_res[x, y]
585  where x is the x-axis of the call
586  y is the y-axis of the call
587  """
588  X = 0
589  Y = 1
590  current_index = 0
591  y = 0
592  x = 0
593  delta_y = 0
594  delta_x = -1
595  movement_res = [X, Y]
596  while current_index != broadCastIndex + 1:
597  movement_res[X] = x * -1
598  movement_res[Y] = y
599  if y == x or (y < 0 and y == -x) or (y > 0 and y == 1 - x):
600  temp = delta_y
601  delta_y = -delta_x
602  delta_x = temp
603  x += delta_x
604  y += delta_y
605  current_index += 1
606  return movement_res
