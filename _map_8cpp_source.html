150 bool Map::_mapCanMove(sf::Vector2f moveMap)
151 {
152  sf::Vector2u windowSize = _window.get()->getSize();
153  sf::FloatRect topTileRect = _tile.at(0).get()->getGlobalBound();
154  sf::FloatRect bottomTileRect = _tile.at(_mapSize.x * _mapSize.y - 1).get()->getGlobalBound();
155  sf::FloatRect leftTileRect = _tile.at((_mapSize.y - 1) * _mapSize.x).get()->getGlobalBound();
156  sf::FloatRect rightTileRect = _tile.at(_mapSize.x - 1).get()->getGlobalBound();
157 
158  if (moveMap.x < 0 && leftTileRect.left > windowSize.x / 2)
159  return false;
160  if (moveMap.x > 0 && rightTileRect.left + rightTileRect.width < windowSize.x / 2)
161  return false;
162  if (moveMap.y < 0 && topTileRect.top > windowSize.y / 2)
163  return false;
164  if (moveMap.y > 0 && bottomTileRect.top + bottomTileRect.height < windowSize.y / 2)
165  return false;
166  return true;
167 }
