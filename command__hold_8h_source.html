378 static inline command_data_t *find_ai_command_end(char *cmd, char *arg)
379 {
380  command_data_t *command = malloc(sizeof(command_data_t));
381  int pos = 0;
382 
383  if (command == NULL)
384  return NULL;
385  for (; ai_command_list_end[pos].name; pos++)
386  if (!strcmp(ai_command_list_end[pos].name, cmd))
387  break;
388  if (!ai_command_list_end[pos].name)
389  return NULL;
390  command->name = ai_command_list_end[pos].name;
391  command->ptr = ai_command_list_end[pos].ptr;
392  command->arg = (!arg) ? NULL : strdup(arg);
393  return command;
394 }
